021import com.fasterxml.jackson.core.JsonGenerator;
022import com.fasterxml.jackson.databind.BeanDescription;
023import com.fasterxml.jackson.databind.JsonSerializer;
024import com.fasterxml.jackson.databind.SerializationConfig;
025import com.fasterxml.jackson.databind.SerializerProvider;
026import com.fasterxml.jackson.databind.ser.BeanSerializerModifier;
027import java.io.IOException;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030
031import static java.util.Objects.requireNonNull;
032
033/**
034 * {@link JSONBean} {@link BeanSerializerModifier} implementation.
035 *
036 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
037 */
038@NoArgsConstructor @ToString
039public class JSONBeanSerializerModifier extends BeanSerializerModifier {
040    @Override
041    public JsonSerializer<?> modifySerializer(SerializationConfig config, BeanDescription description, JsonSerializer<?> serializer) {
042        return ((JSONBean.class.isAssignableFrom(description.getBeanClass()))
043                    ? new SerializerImpl(serializer)
044                    : serializer);
045    }
046
047    @ToString
048    private class SerializerImpl extends JsonSerializer<Object> {
049        private final JsonSerializer<Object> serializer;
050
051        @SuppressWarnings("unchecked")
052        public SerializerImpl(JsonSerializer<?> serializer) {
053            super();
054
055            this.serializer = (JsonSerializer<Object>) requireNonNull(serializer, "serializer");
056        }
057
058        @Override
059        public void serialize(Object value, JsonGenerator generator, SerializerProvider serializers) throws IOException {
060            if (value instanceof JSONBean && ((JSONBean) value).node != null) {
061                generator.writeTree(((JSONBean) value).node);
062            } else {
063                serializer.serialize(value, generator, serializers);
064            }
065        }
066    }
067}
