054/**
055 * Abstract {@link.uri http://ant.apache.org/ Ant} base {@link Task} for
056 * {@link ObjectMapper} tasks.
057 *
058 * {@ant.task}
059 *
060 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
061 */
062@RequiredArgsConstructor(access = PROTECTED)
063public abstract class ObjectMapperTask extends Task implements AnnotatedAntTask,
064                                                               ClasspathDelegateAntTask, ConfigurableAntTask {
065    @Getter @Setter @Accessors(chain = true, fluent = true)
066    private ClasspathUtils.Delegate delegate = null;
067    @Getter @Setter
068    private boolean registerModules = false;
069    private final ArrayList<Setting> settings = new ArrayList<>();
070    @NonNull
071    protected final ObjectMapper mapper;
072
073    /**
074     * No-argument constructor.
075     */
076    protected ObjectMapperTask() { this(new ObjectMapper()); }
077
078    public void addConfiguredConfigure(Setting setting) {
079        settings.add(setting);
080    }
081
082    @Override
083    public void init() throws BuildException {
084        super.init();
085        ClasspathDelegateAntTask.super.init();
086        ConfigurableAntTask.super.init();
087
088        try {
089            if (isRegisterModules()) {
090                mapper.registerModules(ObjectMapper.findModules(getClassLoader()));
091            }
092
093            for (Map.Entry<String,Object> entry :
094                     getProject().getProperties().entrySet()) {
095                if (MAP.containsKey(entry.getKey())) {
096                    ObjectMapperFeature.configure(mapper,
097                                                  MAP.get(entry.getKey()), PropertyHelper.toBoolean(entry.getValue()));
098                }
099            }
100
101            for (Setting setting : settings) {
102                ObjectMapperFeature.configure(mapper, setting.getEnum(), setting.booleanValue());
103            }
104        } catch (BuildException exception) {
105            throw exception;
106        } catch (Throwable throwable) {
107            throw new BuildException(throwable);
108        }
109    }
110
111    @Override
112    public void execute() throws BuildException {
113        super.execute();
114        AnnotatedAntTask.super.execute();
115    }
116
117    /**
118     * {@link ObjectMapper} configuration setting.
119     *
120     * {@bean.info}
121     */
122    @NoArgsConstructor @ToString
123    public static class Setting extends StringAttributeType {
124
125        /**
126         * Method to get the feature {@link Enum}.
127         *
128         * @return      The feature {@link Enum}.
129         */
130        public Enum<?> getEnum() { return MAP.get(getName()); }
131
132        /**
133         * Method to get the feature boolean value.
134         *
135         * @return      The feature boolean value.
136         */
137        public boolean booleanValue() {
138            return (isEmpty(getValue()) ? true : PropertyHelper.toBoolean(getValue()));
139        }
140    }
141
142    /**
143     * {@link.uri http://ant.apache.org/ Ant}
144     * {@link org.apache.tools.ant.Task} to invoke
145     * {@link ObjectMapper#readValue(File,Class)}.
146     *
147     * {@ant.task}
148     */
149    @AntTask("om-read-value")
150    @ToString
151    public static class ReadValue extends ObjectMapperTask {
152        @NotNull @Getter @Setter
153        private File file = null;
154        @NotNull @Getter @Setter
155        private String type = null;
156        @Getter @Setter
157        private String collection = null;
158
159        /**
160         * No-argument constructor.
161         */
162        public ReadValue() { this(new ObjectMapper()); }
163
164        /**
165         * Protected constructor to allow subclasses to specify the
166         * {@link ObjectMapper}.
167         *
168         * @param       mapper  The {@link ObjectMapper}.
169         */
170        protected ReadValue(ObjectMapper mapper) { super(mapper); }
171
172        /**
173         * Method to construct a {@link JavaType} from {@code getType()} and
174         * {@code getCollection()}.
175         *
176         * @return      The {@link JavaType}.
177         *
178         * @throws      Exception       If the {@link JavaType} cannot be
179         *                              constructed.
180         */
181        protected JavaType getJavaType() throws Exception {
182            TypeFactory factory = mapper.getTypeFactory();
183            JavaType type = null;
184
185            if (! isEmpty(getCollection())) {
186                type =
187                    factory.constructCollectionType(getClassForName(getCollection()).asSubclass(Collection.class),
188                                                    getClassForName(getType()));
189            } else {
190                type = factory.constructSimpleType(getClassForName(getType()), new JavaType[] { });
191            }
192
193            return type;
194        }
195
196        @Override
197        public void execute() throws BuildException {
198            super.execute();
199
200            try {
201                Object value = mapper.readValue(getFile(), getJavaType());
202
203                log(String.valueOf(value));
204            } catch (BuildException exception) {
205                throw exception;
206            } catch (Throwable throwable) {
207                throw new BuildException(throwable);
208            }
209        }
210    }
