021import ball.util.EnumLookupMap;
022import com.fasterxml.jackson.core.JsonGenerator;
023import com.fasterxml.jackson.core.JsonParser;
024import com.fasterxml.jackson.databind.DeserializationFeature;
025import com.fasterxml.jackson.databind.MapperFeature;
026import com.fasterxml.jackson.databind.ObjectMapper;
027import com.fasterxml.jackson.databind.SerializationFeature;
028import com.fasterxml.jackson.databind.json.JsonMapper;
029import java.lang.reflect.InvocationTargetException;
030import java.util.SortedMap;
031
032import static java.util.Collections.unmodifiableSortedMap;
033
034/**
035 * Abstract class with static {@link SortedMap} ({@link EnumLookupMap})
036 * member of all {@link ObjectMapper} features.
037 *
038 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
039 */
040public abstract class ObjectMapperFeature {
041
042    /**
043     * The {@link SortedMap} of feature names to their corresponding
044     * {@link Enum}.
045     */
046    public static final SortedMap<String,Enum<?>> MAP =
047        unmodifiableSortedMap(new EnumLookupMap(DeserializationFeature.class,
048                                                JsonGenerator.Feature.class,
049                                                JsonParser.Feature.class,
050                                                MapperFeature.class,
051                                                SerializationFeature.class));
052
053    /**
054     * Static method to configure an {@link JsonMapper.Builder} feature.
055     *
056     * @param   builder         The {@link JsonMapper.Builder} to configure.
057     * @param   feature         The feature {@link Enum}.
058     * @param   value           The {@link Boolean} value.
059     *
060     * @throws  IllegalArgumentException
061     *                          If the feature {@link Enum} is not
062     *                          recognized.
063     */
064    public static void configure(JsonMapper.Builder builder, Enum<?> feature, boolean value) {
065        if (feature instanceof MapperFeature) {
066            builder.configure((MapperFeature) feature, value);
067        } else if (feature instanceof JsonGenerator.Feature) {
068            builder.configure((JsonGenerator.Feature) feature, value);
069        } else if (feature instanceof JsonParser.Feature) {
070            builder.configure((JsonParser.Feature) feature, value);
071        } else if (feature instanceof DeserializationFeature) {
072            builder.configure((DeserializationFeature) feature, value);
073        } else if (feature instanceof SerializationFeature) {
074            builder.configure((SerializationFeature) feature, value);
075        } else {
076            throw new IllegalArgumentException("Unrecognized feature `"
077                                               + feature.getClass().getName() + "." + feature.name() + "'");
078        }
079    }
080
081    /**
082     * Static method to configure an {@link ObjectMapper} feature.
083     *
084     * @param   mapper          The {@link ObjectMapper} to configure.
085     * @param   feature         The feature {@link Enum}.
086     * @param   value           The {@link Boolean} value.
087     *
088     * @throws  IllegalArgumentException
089     *                          If the feature {@link Enum} is not
090     *                          recognized.
091     */
092    public static void configure(ObjectMapper mapper, Enum<?> feature, boolean value) {
093        if (feature instanceof MapperFeature) {
094            /*
095             * mapper.configure((MapperFeature) feature, value);
096             *
097             * is deprecated.  Going forward, the JsonMapper.Builder shoud be configured with this MapperFeature but
098             * attempt the call via reflection for backward compatibility.
099             */
100            try {
101                mapper.getClass()
102                    .getMethod("configure", MapperFeature.class, Boolean.TYPE)
103                    .invoke(mapper, feature, value);
104            } catch (NoSuchMethodException | IllegalAccessException | InvocationTargetException exception) {
105            }
106        } else if (feature instanceof JsonGenerator.Feature) {
107            mapper.configure((JsonGenerator.Feature) feature, value);
108        } else if (feature instanceof JsonParser.Feature) {
109            mapper.configure((JsonParser.Feature) feature, value);
110        } else if (feature instanceof DeserializationFeature) {
111            mapper.configure((DeserializationFeature) feature, value);
112        } else if (feature instanceof SerializationFeature) {
113            mapper.configure((SerializationFeature) feature, value);
114        } else {
115            throw new IllegalArgumentException("Unrecognized feature `"
116                                               + feature.getClass().getName() + "." + feature.name() + "'");
117        }
118    }
119
120    private ObjectMapperFeature() { }
121}
